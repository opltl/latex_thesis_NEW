\begin{equation}
    \text{Nu}=\Delta^{0.3}\cdot(x_1+x_2\cdot\text{Pe}^{x_3})
\end{equation}

\begin{itemize}
\setlength{\parskip}{0pt}
    \item \(x_1=4.75\)
    \item \(x_2=0.0175\)
    \item \(x_3=0.8\)
    \item \(\Delta=\frac{D}{d}=1\div7\)
\end{itemize}

    \text{Nu}=\Delta^{0.3}\cdot(4.75+0.0175\cdot\text{Pe}^{0.8})\\
